        <label for="effort">Effort Score (%):</label>

        <input type="number" id="effort" name="effort" min="0" max="100">

        <label for="improvement">Improvement Score (%):</label>

        <input type="number" id="improvement" name="improvement" min="0" max="100">

        <button onclick="calculateGrade()">Calculate Final Grade</button>

        <p id="finalGrade"></p>

    </div>


    <script>

        function calculateGrade() {

            const participation = parseFloat(document.getElementById('participation').value);

            const effort = parseFloat(document.getElementById('effort').value);

            const improvement = parseFloat(document.getElementById('improvement').value);


            if (isNaN(participation) || isNaN(effort) || isNaN(improvement)) {

                document.getElementById('finalGrade').textContent = "Please enter valid scores for all categories.";

                return;

            }


            const finalGrade = (0.4 * participation) + (0.4 * effort) + (0.2 * improvement);

            document.getElementById('finalGrade').textContent = `Final Grade: ${finalGrade.toFixed(2)}%`;

        }

    </script>
